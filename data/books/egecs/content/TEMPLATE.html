Задача 1

Миша заполнял

# Решение задачи 1
print('x y w z') #шапка таблицы

# 4 цикла для перебора значений 0 и 1 переменных x y w z 
for x in 0,1:
  for y in 0,1:
    for w in 0,1:
      for z in 0,1:
        f = ((x or y) <= z) or (y == w) or z #логическая формула
        if f == 0: #проверяем, что формула вернет нам 0
          print(x,y,w,z) #печатаем значения переменных, если формула вернула 0
